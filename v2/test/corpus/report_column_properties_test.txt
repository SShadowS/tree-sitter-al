{
    dataset
    {
        dataitem(Item; Item)
        {
            column(ItemNo; "No.")
            {
                IncludeCaption = true;
                Caption = 'Item Number';
            }
            column(Description; Description)
            {
                IncludeCaption = false;
            }
            column(UnitPrice; "Unit Price")
            {
                IncludeCaption = true;
                AutoFormatExpression = "Currency Code";
                AutoFormatType = 2;
                DecimalPlaces = 2:5;
            }
            column(Status; Status)
            {
                OptionCaption = 'Open,Released,Closed';
            }
        }
    }
}
