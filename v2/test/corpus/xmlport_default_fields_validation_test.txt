    Direction = Export;
    FieldDelimiter = '<~>';

    TextEncoding = UTF16;
    UseRequestPage = false;

    schema
    {
        textelement(root)

            tableelement(BCPTLogEntry; "BCPT Log Entry")

                fieldelement(EntryNo; BCPTLogEntry."Entry No.") { }
