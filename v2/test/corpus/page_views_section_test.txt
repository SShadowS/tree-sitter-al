pageextension 50000 "Test Views" extends "User List"
{
    views
    {
        addafter(ActiveUsers)
        {
            view(EssentialUsers)
            {
                Caption = 'Essential Users';
                Filters = where("Has Essential Plan" = const(true));
                Visible = true;
            }
            view(PremiumUsers)
            {
                Caption = 'Premium Users';
                Filters = where("Has Premium Plan" = const(true));
                Visible = IsSaaS;
            }
        }
    }
}
