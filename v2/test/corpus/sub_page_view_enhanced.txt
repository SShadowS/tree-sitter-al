pageextension 123 TestExt extends TestPage
{
    layout
    {
        addlast(factboxes)
        {
            part(TestPart; "Test Page")
            {
                SubPageView = sorting("Field1", "Field2") 
                              where("Field3" = const(123), "Field4" = filter('A..Z'));
            }
        }
    }
}
