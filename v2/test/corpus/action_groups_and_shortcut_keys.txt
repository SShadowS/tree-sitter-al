                    Caption = 'Customer Card';

                    PromotedOnly = true;
                    RunObject = page "Customer Card";
                    RunPageLink = "No." = field("Bill-to Customer No.");

                    ToolTip = 'View customer card.';
                }
                action(Statistics)
                {
                    ApplicationArea = All;
                    Caption = 'Statistics';
                    RunPageLink = "No." = field("No.");
                    ShortcutKey = "Ctrl+F7";
                }
            }
        }
    }
}
