    actions
    {
        area(processing)
        {
            action(ImportXML)
            {
                RunObject = xmlport 50100;
            }
            action(RunReport)
            {
                RunObject = report "Customer List";
            }
            action(OpenPage)
            {
                RunObject = page 21;
            }
            action(RunCodeunit)
            {
                RunObject = codeunit "Sales-Post";
            }
            action(OpenTable)
            {
                RunObject = table Customer;
            }
        }
    }
}
