codeunit 50000 "Test"
{
    procedure Test()
    var
        Buffer: Record "Test" temporary;
        Mgt: Codeunit "Mgt";
    begin
#if not CLEAN24
        if not Mgt.IsEnabled() then begin
            Buffer.Reset();
            if Buffer.Find('-') then
                repeat
                    Buffer.Quantity := Buffer.RecordedQty - Buffer.ExpectedQty;
                    Buffer.Modify();
                until Buffer.Next() = 0;
        end else begin
#endif
            Buffer.Reset();
            if Buffer.Find('-') then
                repeat
                    Buffer.Quantity := Buffer.RecordedQty - Buffer.ExpectedQty;
                    Buffer.Modify();
                until Buffer.Next() = 0;
#if not CLEAN24
        end;
#endif
    end;
}
