{
    schema
    {
        textelement(Root)
        {
            tableelement(BCPTSuiteLine; "BCPT Line")
            {
                trigger OnBeforeInsertRecord()
                var
                    BCPTLine: Record "BCPT Line";
                begin
                    BCPTLine.SetAscending("Line No.", true);
                end;
            }
        }
    }
}
