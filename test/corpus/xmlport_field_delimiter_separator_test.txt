{
    Caption = 'Import IRIS to Fault Codes';
    Direction = Import;
    FieldDelimiter = '<None>';
    FieldSeparator = '<TAB>';
    Format = VariableText;
    UseRequestPage = false;

    schema
    {
        textelement(Root)
        {
            tableelement("Fault Code"; "Fault Code")
            {
                XmlName = 'FaultCode';
                fieldelement(Code; "Fault Code".Code)
                {
                }
                fieldelement(Description; "Fault Code".Description)
                {
                }
            }
        }
    }
}
