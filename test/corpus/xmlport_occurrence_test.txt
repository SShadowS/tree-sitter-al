{
    schema
    {
        textelement(Root)
        {
            tableelement(MyTable; "Test Table")
            {
                fieldattribute(Code; MyTable.Code)
                {
                    Occurrence = Required;
                }
                fieldattribute(Description; MyTable.Description)
                {
                    Occurrence = Optional;
                }
            }
        }
    }
}

==================

{
    schema
    {
        textelement(Root)
        {
            tableelement(Table; "Table Name")
            {
                fieldattribute(ID; Table."ID Field")
                {
                    Occurrence = Required;
                    Caption = 'Identifier';
                }
            }
        }
    }
}
