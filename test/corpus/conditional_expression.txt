        x, y: Integer;
        ColumnValues: array[10] of Decimal;
        i: Integer;
        Rec: Record Item;
    begin
        x := y > 0 ? y * 2 : -y;
        ColumnValues[i] *= Rec."Show Opposite Sign" ? -1 : 1;
    end;
