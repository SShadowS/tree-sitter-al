report 38 "Trial Balance by Period"
{
    dataset
    {
        dataitem("Integer"; "Integer")
        {
            column(ColumnValuesAsText_1_; ColumnValuesAsText[1])
            {
                AutoCalcField = false;
            }
            column(ColumnValuesAsText_2_; ColumnValuesAsText[2])
            {
                AutoCalcField = true;
            }
        }
    }
}
