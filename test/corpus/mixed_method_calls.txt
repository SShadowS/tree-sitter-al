{
    var
        Cust: Record Customer;
        Amount: Decimal;
    trigger OnRun()
    begin
        Message('Hello');
        Cust.Get('10000');
        Amount := 100;
        Message('Amount: %1', Amount);
        Cust.SetRange("No.", '10000', '20000');
        if Cust.FindSet() then begin
            Message('Found records');
        end;
    end;
}
