================================================================================
Preprocessor directives in actions section
================================================================================

pageextension 50000 MyPageExt extends "Customer List"
{
    actions
    {
#if not CLEAN24
        addbefore(Category_Process)
        {
            actionref(MyAction_Promoted; MyAction)
            {
                Visible = false;
                ObsoleteState = Pending;
            }
        }
#endif
        addfirst(processing)
        {
            action(NewAction)
            {
                Caption = 'New Action';
                ApplicationArea = All;
            }
        }
    }
}
