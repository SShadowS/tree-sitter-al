{
    dataset
    {
        dataitem(TestData; "Integer")
        {
            DataItemTableView = sorting(Number) where(Number = filter(1 .. 100));
        }
    }
}

    layout
    {
        area(content)
        {
            part("Test Part"; "Test Page")
            {
                SubPageLink = "No." = filter(1 .. 100);
            }
        }
    }
}
