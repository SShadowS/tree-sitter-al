================================================================================
Codeunit with compound assignment operators
================================================================================

codeunit 123 "Test Compound Assignment"
{
    trigger OnRun()
    var
        Counter: Integer;
        Total: Decimal;
        Message: Text;
    begin
        Counter += 1;
        Counter -= 2;
        Total *= 3.5;
        Total /= 2;
        Message += 'Hello';
        
        // Compound assignment with expressions
        Counter += (5 + 3);
        Total += GetValue();
        Message += FormatValue(Counter);
    end;
}
