codeunit 50300 "Auth Handler"
{
    procedure ProcessToken(Token: SecretText): Boolean
    var
        IsValid: Boolean;
    begin
        if Token.IsEmpty() then
            exit(false);
        
        IsValid := ValidateToken(Token);
        exit(IsValid);
    end;
}
