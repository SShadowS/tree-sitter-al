codeunit 123 "Test Range Case"
{
    procedure TestRangeCase()
    var
        Code: array[10] of Code[20];
        i: Integer;
    begin
        case Code[i] of
            'A'..'Z', '0'..'9':
                exit;
            'Æ':
                exit;
        end;
    end;
}
