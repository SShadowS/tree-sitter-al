{
    fields
    {
        field(1; "Complex Field"; Code[20])
        {
            TableRelation = IF (Type = CONST(1)) Table1
            ELSE
            IF (Type = CONST(2)) Table2
            ELSE
            IF (Type = CONST(3)) Table3
            ELSE
            #if NEWVERSION
            IF (Type = CONST(4)) Table4 WHERE(Field = CONST(Value));
            #else
            IF (Type = CONST(4)) Table5;
            #endif;

            trigger OnValidate()
            begin
                Message('Validated');
            end;
        }
    }
}
