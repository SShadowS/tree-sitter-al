reportextension 99000858 "Mfg. Get Demand To Reserve" extends "Get Demand To Reserve"
{
    dataset
    {
        addafter(TransferOrderLine)
        {
            dataitem(ProdOrderComponent; "Prod. Order Component")
            {
                DataItemTableView = sorting(Status, "Prod. Order No.")
                                    where(Status = const(Released));
                trigger OnPreDataItem()
                begin
                    if not Filter then
                        CurrReport.Break();
                end;
            }
        }
    }
}

================================================================================
Report extension with multiple dataset modifications
================================================================================

reportextension 50100 "Extended Report" extends "Base Report"
{
    dataset
    {
        addfirst
        {
            dataitem(Header; "Sales Header")
            {
                RequestFilterFields = "No.";
            }
        }
        
        addafter(Customer)
        {
            dataitem(SalesLine; "Sales Line")
            {
                DataItemLink = "Document No." = field("No.");
            }
        }
        
        addlast
        {
            dataitem(Footer; "Custom Footer")
            {
                MaxIteration = 1;
            }
        }
    }
}
