query 50102 "Mixed Var Query"
{
    elements
    {
        dataitem(Vendor; Vendor)
        {
            column(No; "No.")
            {
            }
        }
    }

    var
        GlobalText: Text;
        GlobalCounter: Integer;

    procedure ProcessData()
    var
        LocalVendor: Record Vendor;
        LocalText: Text[50];
    begin
        GlobalCounter += 1;
        LocalText := GlobalText;
    end;

    trigger OnBeforeOpen()
    begin
        GlobalCounter := 0;
    end;
}
