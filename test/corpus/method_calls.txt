codeunit 50100 MyCodeunit
{
    var
        Customer: Record Customer;
    trigger OnRun()
    begin
        if Customer.Get('10000') then
            Message('Found customer');
        
        if Customer.FindSet() then begin
            repeat
                if Customer.Next() then
                    Message('Next customer');
            until Customer."No." = '';
        end;
    end;
}
