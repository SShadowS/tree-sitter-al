controladdin TestAddin
{
    RequestedHeight = 100;
    
#if CLEAN30
    procedure NewMethod();
#else
    #if not CLEAN25
        procedure OldMethod();
    #else
        procedure VeryOldMethod();
    #endif
#endif
}
