Table extension with field and modify statements combined
================================================================================

tableextension 50001 "Mixed Extension" extends "Base Table"
{
    fields
    {
        field(50000; "New Field"; Text[50])
        {
            Caption = 'New Field';
        }
        modify("Existing Field")
        {
            Caption = 'Updated Caption';
        }
    }
}
