{
    fields
    {
        field(1; "No."; Code[20]) { }
        field(2; "Creation Date"; Date) { }
        field(3; SystemCreatedAt; DateTime) { }
    }
    
    fieldgroups
    {
#if not CLEAN24
        fieldgroup(DropDown; "No.", "Creation Date")
        {
        }
#else
        fieldgroup(DropDown; "No.", SystemCreatedAt)
        {
        }
#endif
    }
}
